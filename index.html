    function shuffle(array) {

      const copy = [...array];

      for (let i = copy.length - 1; i > 0; i--) {

        const j = Math.floor(Math.random() * (i + 1));

        [copy[i], copy[j]] = [copy[j], copy[i]];

      }

      return copy;

    }


    function createDragTasks() {

      Object.keys(dragData).forEach(taskId => {

        const pool = document.getElementById("pool" + taskId);

        pool.innerHTML = "";

        shuffle(dragData[taskId]).forEach((item, index) => {

          const div = document.createElement("div");

          div.className = "draggable";

          div.draggable = true;

          div.textContent = item.text;

          div.dataset.zone = item.zone;

          div.id = `drag-${taskId}-${index}-${Math.random().toString(36).slice(2)}`;

          div.addEventListener("dragstart", dragStart);

          pool.appendChild(div);

        });

      });


      document.querySelectorAll(".drop-zone, .drag-pool").forEach(zone => {

        zone.addEventListener("dragover", dragOver);

        zone.addEventListener("dragleave", dragLeave);

        zone.addEventListener("drop", dropItem);

      });

    }


    function dragStart(e) {

      e.dataTransfer.setData("text/plain", e.target.id);

    }


    function dragOver(e) {

      e.preventDefault();

      if (e.currentTarget.classList.contains("drop-zone")) {

        e.currentTarget.classList.add("over");

      }

    }


    function dragLeave(e) {

      e.currentTarget.classList.remove("over");

    }


    function dropItem(e) {

      e.preventDefault();

      const id = e.dataTransfer.getData("text/plain");

      const item = document.getElementById(id);

      e.currentTarget.classList.remove("over");

      if (item) e.currentTarget.appendChild(item);

    }


    function checkDragTask(taskNumber) {

      if (solved[taskNumber - 1]) return;


      const card = document.getElementById("card" + taskNumber);

      const zones = card.querySelectorAll(".drop-zone");

      let totalItems = 0;

      let correct = true;


      zones.forEach(zone => {

        const expected = zone.dataset.zone;

        const items = zone.querySelectorAll(".draggable");

        totalItems += items.length;


        items.forEach(item => {

          if (item.dataset.zone !== expected) correct = false;

        });

      });


      const needed = dragData[taskNumber].length;

      if (totalItems !== needed) correct = false;


      if (correct) {

        solveTask(taskNumber);

      } else {

        showFeedback(taskNumber, false, "Noch nicht richtig. Prüfe die Kategorien und versuche es noch einmal.");

      }

    }


    function createSortTask() {

      const list = document.getElementById("sortList");

      list.innerHTML = "";


      shuffle(sortCorrect).forEach(text => {

        const item = document.createElement("div");

        item.className = "sort-item";

        item.dataset.value = text;

        item.innerHTML = `

          <span>${text}</span>

          <div class="sort-controls">

            <button type="button" onclick="moveItem(this, -1)" aria-label="nach oben">

              <i class="ti ti-arrow-up"></i>

            </button>

            <button type="button" onclick="moveItem(this, 1)" aria-label="nach unten">

              <i class="ti ti-arrow-down"></i>

            </button>

          </div>

        `;

        list.appendChild(item);

      });

    }


    function moveItem(button, direction) {

      const item = button.closest(".sort-item");

      const list = document.getElementById("sortList");


      if (direction === -1 && item.previousElementSibling) {

        list.insertBefore(item, item.previousElementSibling);

      }


      if (direction === 1 && item.nextElementSibling) {

        list.insertBefore(item.nextElementSibling, item);

      }

    }

    function solveTask(taskNumber) {

      solved[taskNumber - 1] = true;


      const letter = letters[taskNumber - 1];

      const badge = document.getElementById("badge" + taskNumber);

      const card = document.getElementById("card" + taskNumber);

      const slot = document.getElementById("slot" + (taskNumber - 1));


      badge.textContent = letter;

      badge.classList.add("revealed");


      slot.textContent = letter;

      slot.classList.add("revealed");


      card.classList.add("solved");


      showFeedback(taskNumber, true, `¡Correcto! Has conseguido la letra ${letter}.`);


      updateProgress();

    }


    function showFeedback(taskNumber, isOk, message) {

      const feedback = document.getElementById("feedback" + taskNumber);

      feedback.textContent = isOk ? "✓ " + message : "✗ " + message;

      feedback.className = "feedback " + (isOk ? "ok" : "bad");

    }


    function updateProgress() {

      const count = solved.filter(Boolean).length;

      document.getElementById("progressText").textContent = `${count} de 5 candados resueltos`;

      document.getElementById("progressFill").style.width = `${count * 20}%`;


      if (count === 5) {

        document.getElementById("finalCard").classList.add("visible");

        document.getElementById("finalCard").scrollIntoView({ behavior: "smooth", block: "center" });

      }

    }


    function checkFinalCode() {

      const input = document.getElementById("finalInput").value.trim().toUpperCase();

      const feedback = document.getElementById("finalFeedback");

      const success = document.getElementById("successBox");


      if (input === CODE_WORD) {

        feedback.textContent = "✓ Código correcto.";

        feedback.className = "feedback ok";

        success.classList.add("visible");

        feedback.className = "feedback bad";

        success.classList.remove("visible");

      }

    }

        setTimeout(() => confetti.remove(), 2400);

      }

    }
